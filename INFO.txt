--------------- КОМАНДЫ ---------------

CREATE TABLE <название таблицы> - создать таблицу.

INSERT INTO <название таблицы> <элемент> - добавить элемент в таблицу.

DELETE FROM <название таблицы> WHERE <название таблицы>.колонка<номер колонки> = <значение> - удалить строку по критерию.

SELECT * FROM <название таблицы1> <название таблицы2> - вывести декартово произведение таблиц.

SELECT <название таблицы>.колонка<номер колонки> .... (до 5 таблиц через запятую) FROM <названия через запятую> - вывести колонки из таблиц.

SELECT <название таблицы>.колонка<номер колонки> FROM <название таблицы> WHERE <название таблицы>.колонка<номер колонки> = <значение> - вывести строку по критерию.

SELECT <название таблицы>.колонка<номер колонки> .... (до 2 таблиц через запятую) FROM <название таблиц через запятую> WHERE <название таблицы>.колонка<номер колонки> = <значение> - вывести декартово произведение строк по критериям.


ВАЖНО!!! В командах с критериями возможно составление логического выражения с операторами OR и AND.
